15def get_number_of_processes(parallel_scheme: str, nprocs: int = None): 

16 """This function works out how many processes have been set 

17 by the paralellisation system called 'parallel_scheme' 

18 """ 

19 if nprocs is None: 

20 if parallel_scheme == "multiprocessing": 

21 return 1 

22 elif parallel_scheme == "mpi4py": 

23 from mpi4py import MPI 

24 comm = MPI.COMM_WORLD 

25 nprocs = comm.Get_size() 

26 return nprocs 

27 elif parallel_scheme == "scoop": 

28 raise ValueError( 

29 f"You must specify the number of processes for " 

30 f"scoop to parallelise over") 

31 else: 

32 raise ValueError( 

33 f"You must specify the number of processes to " 

34 f"use for parallel scheme '{parallel_scheme}'") 

35 

36 if parallel_scheme == "mpi4py": 

37 from mpi4py import MPI 

38 comm = MPI.COMM_WORLD 

39 n = comm.Get_size() 

40 

41 if n < nprocs: 

42 return n 

43 else: 

44 return nprocs 

45 elif parallel_scheme == "scoop": 

46 return 4 

47 elif parallel_scheme == "multiprocessing": 

48 return nprocs 

49 else: 

50 raise ValueError( 

51 f"Unrecognised parallelisation scheme {parallel_scheme}")

54@_contextmanager 

55def redirect_output(outdir): 

56 """Nice way to redirect stdout and stderr - thanks to 

57 Emil Stenström in 

58 https://stackoverflow.com/questions/6735917/redirecting-stdout-to-nothing-in-python 

59 """ 

60 new_out = open(_os.path.join(outdir, "output.txt"), "w") 

61 old_out = _sys.stdout 

62 _sys.stdout = new_out 

63 

64 new_err = open(_os.path.join(outdir, "output.err"), "w") 

65 old_err = _sys.stderr 

66 _sys.stderr = new_err 

67 

68 try: 

69 yield new_out 

70 finally: 

71 _sys.stdout = old_out 

72 _sys.stderr = old_err 

73 new_out.close() 

74 new_err.close()

77def run_models(network: Network, variables: VariableSets, 

78 population: Population, 

79 nprocs: int, nthreads: int, seed: int, 

80 nsteps: int, output_dir: OutputFiles, 

81 iterator: str, extractor: str, 

82 profile: bool, parallel_scheme: str): 

83 """Run all of the models on the passed Network that are described 

84 by the passed VariableSets 

85 

86 Parameters 

87 ---------- 

88 network: Network 

89 The network to model 

90 variables: VariableSets 

91 The sets of VariableSet that represent all of the model 

92 runs to perform 

93 population: Population 

94 The initial population for all of the model runs. This also 

95 contains the starting date and day for the model outbreak 

96 nprocs: int 

97 The number of model runs to perform in parallel 

98 nthreads: int 

99 The number of threads to parallelise each model run over 

100 seed: int 

101 Random number seed which is used to generate random seeds 

102 for all model runs 

103 nsteps: int 

104 The maximum number of steps to perform for each model - this 

105 will run until the outbreak is over if this is None 

106 output_dir: OutputFiles 

107 The OutputFiles that represents the directory in which all 

108 output should be placed 

109 iterator: str 

110 Iterator to load that will be used to iterate the outbreak 

111 extractor: str 

112 Extractor to load that will be used to extract information 

113 profile: bool 

114 Whether or not to profile the model run and print out live 

115 timing (useful for performance debugging) 

116 parallel_scheme: str 

117 Which parallel scheme (multiprocessing, mpi4py or scoop) to use 

118 to run multiple model runs in parallel 

119 """ 

120 

121 # this variable is used to pick out some of the additional seeds? 

122 s = -1 

123 

124 if len(variables) == 1: 

125 # no need to do anything complex - just a single run 

126 params = network.params.set_variables(variables[0]) 

127 

128 network.update(params, profile=profile) 

129 

130 trajectory = network.run(population=population, seed=seed, 

131 s=s, nsteps=nsteps, 

132 output_dir=output_dir, 

133 iterator=iterator, 

134 extractor=extractor, 

135 profile=profile, 

136 nthreads=nthreads) 

137 

138 return [(variables[0], trajectory)] 

139 

140 # generate the random number seeds for all of the jobs 

141 # (for testing, we will use the same seed so that I can check 

142 # that they are all working) 

143 seeds = [] 

144 

145 if seed == 0: 

146 # this is a special mode that a developer can use to force 

147 # all jobs to use the same random number seed (15324) that 

148 # is used for comparing outputs. This should NEVER be used 

149 # for production code 

150 print("** WARNING: Using special mode to fix all random number") 

151 print("** WARNING: seeds to 15324. DO NOT USE IN PRODUCTION!!!") 

152 

153 for i in range(0, len(variables)): 

154 seeds.append(15324) 

155 

156 else: 

157 from ._ran_binomial import seed_ran_binomial, ran_int 

158 rng = seed_ran_binomial(seed) 

159 

160 # seed the rngs used for the sub-processes using this rng 

161 for i in range(0, len(variables)): 

162 seeds.append(ran_int(rng, 10000, 99999999)) 

163 

164 # set the output directories for all of the jobs - this is based 

165 # on the fingerprint, so should be unique for each job 

166 outdirs = [] 

167 

168 for v in variables: 

169 f = v.fingerprint(include_index=True) 

170 d = _os.path.join(output_dir.get_path(), f) 

171 outdirs.append(d) 

172 

173 outputs = [] 

174 

175 print(f"\nRunning {len(variables)} jobs using {nprocs} process(es)") 

176 

177 if nprocs == 1: 

178 # no need to use a pool, as we will repeat this calculation 

179 # several times 

180 for i, variable in enumerate(variables): 

181 seed = seeds[i] 

182 outdir = outdirs[i] 

183 

184 with output_dir.open_subdir(outdir) as subdir: 

185 print(f"\nRunning parameter set {i+1} of {len(variables)} " 

186 f"using seed {seed}") 

187 print(f"All output written to {subdir.get_path()}") 

188 

189 with redirect_output(subdir.get_path()): 

190 print(f"Running variable set {i+1}") 

191 print(f"Variables: {variable}") 

192 print(f"Random seed: {seed}") 

193 print(f"nthreads: {nthreads}") 

194 

195 # no need to do anything complex - just a single run 

196 params = network.params.set_variables(variable) 

197 

198 network.update(params, profile=profile) 

199 

200 output = network.run(population=population, seed=seed, 

201 s=s, nsteps=nsteps, 

202 output_dir=subdir, 

203 iterator=iterator, 

204 extractor=extractor, 

205 profile=profile, 

206 nthreads=nthreads) 

207 

208 outputs.append((variable, output)) 

209 

210 print(f"Completed job {i+1} of {len(variables)}") 

211 print(variable) 

212 print(output[-1]) 

213 # end of OutputDirs context manager 

214 # end of loop over variable sets 

215 else: 

216 from ._worker import run_worker 

217 

218 # create all of the parameters and options to run 

219 arguments = [] 

220 

221 max_nodes = network.nnodes + 1 

222 max_links = max(network.nlinks, network.plinks) + 1 

223 

224 for i, variable in enumerate(variables): 

225 seed = seeds[i] 

226 outdir = outdirs[i] 

227 

228 arguments.append({ 

229 "params": network.params.set_variables(variable), 

230 "options": {"seed": seed, 

231 "output_dir": outdir, 

232 "auto_bzip": output_dir.auto_bzip(), 

233 "population": population, 

234 "s": s, 

235 "nsteps": nsteps, 

236 "iterator": iterator, 

237 "extractor": extractor, 

238 "profile": profile, 

239 "nthreads": nthreads, 

240 "max_nodes": max_nodes, 

241 "max_links": max_links} 

242 }) 

243 

244 if parallel_scheme == "multiprocessing": 

245 # run jobs using a multiprocessing pool 

246 print("\nRunning jobs in parallel using a multiprocessing pool...") 

247 from multiprocessing import Pool 

248 with Pool(processes=nprocs) as pool: 

249 results = pool.map(run_worker, arguments) 

250 

251 for i, result in enumerate(results): 

252 print(f"\nCompleted job {i+1} of {len(variables)}") 

253 print(variables[i]) 

254 print(result[-1]) 

255 outputs.append((variables[i], result)) 

256 

257 elif parallel_scheme == "mpi4py": 

258 # run jobs using a mpi4py pool 

259 print("\nRunning jobs in parallel using a mpi4py pool...") 

260 from mpi4py import futures 

261 with futures.MPIPoolExecutor(max_workers=nprocs) as pool: 

262 results = pool.map(run_worker, arguments) 

263 

264 for i, result in enumerate(results): 

265 print(f"\nCompleted job {i+1} of {len(variables)}") 

266 print(variables[i]) 

267 print(result[-1]) 

268 outputs.append((variables[i], result)) 

269 

270 elif parallel_scheme == "scoop": 

271 # run jobs using a scoop pool 

272 print("\nRunning jobs in parallel using a scoop pool...") 

273 from scoop import futures 

274 

275 results = futures.map(run_worker, arguments) 

276 

277 for i, result in enumerate(results): 

278 print(f"\nCompleted job {i+1} of {len(variables)}") 

279 print(variables[i]) 

280 print(result[-1]) 

281 outputs.append((variables[i], result)) 

282 

283 else: 

284 raise ValueError(f"Unrecognised parallelisation scheme " 

285 f"{parallel_scheme}.") 

286 

287 return outputs
