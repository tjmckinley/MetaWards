18def get_parallel_scheme(): 

19 """This function tries to work out which of the different parallel 

20 methods we should use to distribute work between multiple processes. 

21 

22 Detected schemes are "mpi4py", "scoop", and if none of these 

23 are found, then "multiprocessing" 

24 """ 

25 import sys 

26 

27 if "mpi4py" in sys.modules: 

28 return "mpi4py" 

29 elif "scoop" in sys.modules: 

30 return "scoop" 

31 else: 

32 return "multiprocessing"

419def scoop_supervisor(hostfile, args): 

420 """Function used by the scoop supervisor to get the information needed to 

421 form the scoop call to run a scoop version of the program 

422 """ 

423 import os 

424 import sys 

425 from metawards.utils import Console 

426 

427 Console.print("RUNNING A SCOOP PROGRAM") 

428 

429 cores_per_node = get_cores_per_node(args) 

430 

431 Console.print( 

432 f"Will run jobs assuming {cores_per_node} cores per compute node") 

433 

434 # based on the number of threads requested and the number of cores 

435 # per node, we can work out the number of scoop processes to start, 

436 # and can write a hostfile that will create the right layout 

437 nthreads = get_threads_per_task(args) 

438 

439 Console.print(f"Will use {nthreads} OpenMP threads per model run...") 

440 

441 tasks_per_node = int(cores_per_node / nthreads) 

442 

443 Console.print(f"...meaning that the number of model runs per node will be " 

444 f"{tasks_per_node}") 

445 

446 # Next, read the hostfile to get a unique list of hostnames 

447 hostnames = {} 

448 

449 with open(hostfile, "r") as FILE: 

450 line = FILE.readline() 

451 while line: 

452 hostname = line.strip() 

453 if len(hostname) > 0: 

454 hostnames[hostname] = 1 

455 line = FILE.readline() 

456 

457 hostnames = list(hostnames.keys()) 

458 hostnames.sort() 

459 

460 Console.print(f"Number of compute nodes equals {len(hostnames)}") 

461 Console.print(", ".join(hostnames)) 

462 

463 # how many tasks can we perform in parallel? 

464 nprocs = tasks_per_node * len(hostnames) 

465 

466 if args.nprocs: 

467 if nprocs != args.nprocs: 

468 Console.warning( 

469 f"You are using a not-recommended number of " 

470 f"processes {args.nprocs} for the cluster {nprocs}.") 

471 

472 nprocs = args.nprocs 

473 

474 Console.print( 

475 f"Total number of parallel processes to run will be {nprocs}") 

476 Console.print(f"Total number of cores in use will be {nprocs*nthreads}") 

477 

478 # Now write a new hostfile that round-robins the MPI tasks over 

479 # the nodes for 'tasks_per_node' runs 

480 hostfile = f"_metawards_hostfile_{os.getpid()}" 

481 Console.print(f"Writing hostfile to {hostfile}") 

482 

483 with open(hostfile, "w") as FILE: 

484 i = 0 

485 while i < nprocs: 

486 for hostname in hostnames: 

487 FILE.write(hostname + "\n") 

488 i += 1 

489 

490 if i == nprocs: 

491 break 

492 

493 # now craft the scoop command that will use this hostfile to 

494 # run the job - remember to pass the option to stop the main process 

495 # attempting to become a supervisor itself... 

496 

497 import subprocess 

498 import shlex 

499 

500 pyexe = sys.executable 

501 script = os.path.abspath(sys.argv[0]) 

502 args = " ".join(sys.argv[1:]) 

503 

504 # also need to tell the main program the number of processes 

505 # as it can't work it out itself 

506 cmd = f"{pyexe} -m scoop --hostfile {hostfile} -n {nprocs} " \ 

507 f"{script} --already-supervised {args} --nprocs {nprocs}" 

508 

509 Console.print("Executing scoop job using") 

510 Console.command(cmd) 

511 

512 try: 

513 args = shlex.split(cmd) 

514 subprocess.run(args).check_returncode() 

515 except Exception as e: 

516 Console.error("ERROR: Something went wrong!") 

517 Console.error(f"{e.__class__}: {e}") 

518 sys.exit(-1) 

519 

520 # clean up the hostfile afterwards... (we leave it if something 

521 # went wrong as it may help debugging) 

522 os.unlink(hostfile) 

523 

524 Console.print("Scoop processes completed successfully")

655

657 runs and manages the jobs 

658 

659 2. If this is a worker process, then it starts up and waits for work 

660 

661 3. If this is a supervisor process, then it query the job scheduling
