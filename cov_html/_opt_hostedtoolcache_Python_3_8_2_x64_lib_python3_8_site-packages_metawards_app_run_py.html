18def get_parallel_scheme(): 

19 """This function tries to work out which of the different parallel 

20 methods we should use to distribute work between multiple processes. 

21 

22 Detected schemes are "mpi4py", "scoop", and if none of these 

23 are found, then "multiprocessing" 

24 """ 

25 import sys 

26 

27 if "mpi4py" in sys.modules: 

28 return "mpi4py" 

29 elif "scoop" in sys.modules: 

30 return "scoop" 

31 else: 

32 return "multiprocessing"

309def scoop_supervisor(hostfile, args): 

310 """Function used by the scoop supervisor to get the information needed to 

311 form the scoop call to run a scoop version of the program 

312 """ 

313 import os 

314 import sys 

315 print("RUNNING A SCOOP PROGRAM") 

316 

317 outdir = args.output 

318 

319 if not os.path.exists(outdir): 

320 os.mkdir(outdir) 

321 

322 cores_per_node = get_cores_per_node(args) 

323 

324 print(f"Will run jobs assuming {cores_per_node} cores per compute node") 

325 

326 # based on the number of threads requested and the number of cores 

327 # per node, we can work out the number of scoop processes to start, 

328 # and can write a hostfile that will create the right layout 

329 nthreads = get_threads_per_task(args) 

330 

331 print(f"Will use {nthreads} OpenMP threads per model run...") 

332 

333 tasks_per_node = int(cores_per_node / nthreads) 

334 

335 print(f"...meaning that the number of model runs per node will be " 

336 f"{tasks_per_node}") 

337 

338 # Next, read the hostfile to get a unique list of hostnames 

339 hostnames = {} 

340 

341 with open(hostfile, "r") as FILE: 

342 line = FILE.readline() 

343 while line: 

344 hostname = line.strip() 

345 hostnames[hostname] = 1 

346 line = FILE.readline() 

347 

348 hostnames = list(hostnames.keys()) 

349 hostnames.sort() 

350 

351 print(f"Number of compute nodes equals {len(hostnames)}") 

352 print(", ".join(hostnames)) 

353 

354 # how many tasks can we perform in parallel? 

355 nprocs = tasks_per_node * len(hostnames) 

356 

357 if args.nprocs: 

358 if nprocs != args.nprocs: 

359 print(f"WARNING: You are using a not-recommended number of " 

360 f"processes {args.nprocs} for the cluster {nprocs}.") 

361 

362 nprocs = args.nprocs 

363 

364 print(f"Total number of parallel processes to run will be {nprocs}") 

365 print(f"Total number of cores in use will be {nprocs*nthreads}") 

366 

367 # Now write a new hostfile that round-robins the MPI tasks over 

368 # the nodes for 'tasks_per_node' runs 

369 hostfile = os.path.join(outdir, "hostfile") 

370 print(f"Writing hostfile to {hostfile}") 

371 

372 with open(hostfile, "w") as FILE: 

373 i = 0 

374 while i < nprocs: 

375 for hostname in hostnames: 

376 FILE.write(hostname + "\n") 

377 i += 1 

378 

379 if i == nprocs: 

380 break 

381 

382 # now craft the scoop command that will use this hostfile to 

383 # run the job - remember to pass the option to stop the main process 

384 # attempting to become a supervisor itself... 

385 

386 import subprocess 

387 import shlex 

388 

389 pyexe = sys.executable 

390 script = os.path.abspath(sys.argv[0]) 

391 args = " ".join(sys.argv[1:]) 

392 

393 # also need to tell the main program the number of processes 

394 # as it can't work it out itself 

395 cmd = f"{pyexe} -m scoop --hostfile {hostfile} -n {nprocs} " \ 

396 f"{script} --already-supervised {args} --nprocs {nprocs}" 

397 

398 print(f"Executing scoop job using '{cmd}'") 

399 

400 try: 

401 args = shlex.split(cmd) 

402 subprocess.run(args).check_returncode() 

403 except Exception as e: 

404 print("ERROR: Something went wrong!") 

405 print(f"{e.__class__}: {e}") 

406 sys.exit(-1) 

407 

408 print("Scoop processes completed successfully")

530def cli(): 

531 """Main function for the command line interface. This does one of three 

532 things: 

533 

534 1. If this is the main process, then it parses the arguments and 

535 runs and manages the jobs 

536 

537 2. If this is a worker process, then it starts up and waits for work 

538 

539 3. If this is a supervisor process, then it query the job scheduling 

540 system for information about the compute nodes to use, and will then 

541 set up and run a manager (main) process that will use those 

542 nodes to run the jobs 

543 """ 

544 

545 # get the parallel scheme now before we import any other modules 

546 # so that it is clear if mpi4py or scoop (or another parallel module) 

547 # has been imported via the required "-m module" syntax 

548 parallel_scheme = get_parallel_scheme() 

549 

550 if parallel_scheme == "mpi4py": 

551 from mpi4py import MPI 

552 comm = MPI.COMM_WORLD 

553 nprocs = comm.Get_size() 

554 rank = comm.Get_rank() 

555 

556 if rank != 0: 

557 # this is a worker process, so should not do anything 

558 # more until it is given work in the pool 

559 print(f"Starting worker process {rank+1} of {nprocs-1}...") 

560 return 

561 else: 

562 print("Starting main process...") 

563 

564 elif parallel_scheme == "scoop": 

565 print("STARTING SCOOP PROCESS") 

566 

567 import sys 

568 

569 args, parser = parse_args() 

570 

571 if not args.already_supervised: 

572 hostfile = get_hostfile(args) 

573 if hostfile: 

574 # The user has asked to run a parallel job - this means that this 

575 # process is the parallel supervisor 

576 if args.mpi: 

577 mpi_supervisor(hostfile, args) 

578 return 

579 elif args.scoop: 

580 scoop_supervisor(hostfile, args) 

581 return 

582 

583 # neither is preferred - if scoop is installed then use that 

584 try: 

585 import scoop 

586 have_scoop = True 

587 except Exception: 

588 have_scoop = False 

589 

590 if have_scoop: 

591 scoop_supervisor(hostfile, args) 

592 return 

593 

594 # do we have MPI? 

595 try: 

596 import mpi4py 

597 have_mpi4py = True 

598 except Exception: 

599 have_mpi4py = False 

600 

601 if have_mpi4py: 

602 mpi_supervisor(hostfile, args) 

603 return 

604 

605 # we don't have any other option, just keep going and 

606 # use multiprocessing - in this case we don't need a 

607 # supervisor and this is the main process 

608 

609 # This is now the code for the main process 

610 

611 # WE NEED ONE OF these listed options; 

612 should_run = False 

613 

614 for arg in [args.input, args.repeats, args.disease, args.additional]: 

615 if arg is not None: 

616 should_run = True 

617 break 

618 

619 if not should_run: 

620 parser.print_help(sys.stdout) 

621 sys.exit(0) 

622 

623 if args.repeats is None: 

624 args.repeats = 1 

625 

626 # import the parameters here to speed up the display of help 

627 from metawards import Parameters, Network, Population, get_version_string 

628 

629 # print the version information first, so that there is enough 

630 # information to enable someone to reproduce this run 

631 print(get_version_string()) 

632 

633 # also print the full command line used for this job 

634 print(f"Command used to run this job:\n{' '.join(sys.argv)}\n") 

635 

636 if args.input: 

637 # get the line numbers of the input file to read 

638 if args.line is None or len(args.line) == 0: 

639 linenums = None 

640 print(f"Using parameters from all lines of {args.input}") 

641 else: 

642 from metawards.utils import string_to_ints 

643 linenums = string_to_ints(args.line) 

644 

645 if len(linenums) == 0: 

646 print(f"You cannot read no lines from {args.input}?") 

647 sys.exit(-1) 

648 elif len(linenums) == 1: 

649 print(f"Using parameters from line {linenums[0]} of " 

650 f"{args.input}") 

651 else: 

652 print(f"Using parameters from lines {linenums} of " 

653 f"{args.input}") 

654 

655 from metawards import VariableSets, VariableSet 

656 variables = VariableSets.read(filename=args.input, 

657 line_numbers=linenums) 

658 else: 

659 from metawards import VariableSets, VariableSet 

660 # create a VariableSets with one null VariableSet 

661 variables = VariableSets() 

662 variables.append(VariableSet()) 

663 

664 nrepeats = args.repeats 

665 

666 if nrepeats is None or nrepeats < 1: 

667 nrepeats = 1 

668 

669 if nrepeats == 1: 

670 print("Performing a single run of each set of parameters") 

671 else: 

672 print(f"Performing {nrepeats} runs of each set of parameters") 

673 

674 variables = variables.repeat(nrepeats) 

675 

676 # working out the number of processes and threads... 

677 from metawards.utils import guess_num_threads_and_procs 

678 (nthreads, nprocs) = guess_num_threads_and_procs( 

679 njobs=len(variables),
